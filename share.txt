import socket, ssl, sys, hashlib, uuid, base64
from datetime import datetime
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend

KEY_STATIC = "a3(D3$@fK3"
ENCRYPTED_SECRET_KEY = "USJJ2ecM0/T8zLBzfsOec7KB1jIBUXcypIfvedawppk="
IV_ZERO = b"\x00" * 16


def get_key(pw):
    if len(pw.encode("utf-8")) < 24:
        return pw.ljust(24).encode("utf-8")
    return pw[:24].encode("utf-8")


def aes_crypt(data, pw, encrypt=True):
    key = get_key(pw if pw else KEY_STATIC)
    cipher = Cipher(algorithms.AES(key), modes.CBC(IV_ZERO), backend=default_backend())
    op = cipher.encryptor() if encrypt else cipher.decryptor()
    if encrypt:
        pad = 16 - (len(data) % 16)
        data = data + bytes([pad] * pad)
        return op.update(data) + op.finalize()
    else:
        pt = op.update(data) + op.finalize()
        return pt[:-pt[-1]]


def decrypt_secret():
    return aes_crypt(base64.b64decode(ENCRYPTED_SECRET_KEY.replace(" ", "+")), KEY_STATIC, False)


def double_encrypt(plaintext, guid=None):
    guid = guid or str(uuid.uuid4()).replace("-", "")[:16]
    first = base64.b64encode(aes_crypt(plaintext.encode("ascii"), guid + KEY_STATIC)).decode()
    sk = first[:12]
    second = base64.b64encode(aes_crypt(guid.encode(), sk + KEY_STATIC)).decode()
    return first + "|" + second


def sha256_hex(text):
    return hashlib.sha256(text.encode("utf-8")).hexdigest().upper()


def connect(ip, port, tls=True):
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.settimeout(10)
    s.connect((ip, port))
    if tls:
        ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
        ctx.check_hostname = False
        ctx.verify_mode = ssl.CERT_NONE
        s = ctx.wrap_socket(s, server_hostname=ip)
    return s


def send_recv(s, payload):
    s.sendall(payload.encode("utf-8"))
    chunks = []
    try:
        while True:
            chunks.append(s.recv(65536))
    except socket.timeout:
        pass
    return b"".join(chunks).decode("utf-8", errors="replace")
